      class _AddMarketRow extends StatefulWidget {
      @override
      State<_AddMarketRow> createState() => _AddMarketRowState();
      }
      class _AddMarketRowState extends State<_AddMarketRow> {
      final loc = TextEditingController();
      final price = TextEditingController();
      @override
      Widget build(BuildContext context) {
      final store = InheritedStore.of(context);
      // Only stakeholder or farmer can add in demo, but we allow any role for demo simplicity.
      return Row(
      children: [
      Expanded(child: TextField(controller: loc, decoration: const InputDecoration(hintText: "লোকেশন"))),
      const SizedBox(width: 8),
      Expanded(child: TextField(controller: price, keyboardType: TextInputType.number, decoration: const InputDecoration(hintText: "দাম"))),
      const SizedBox(width: 8),
      ElevatedButton(
      onPressed: () {
      if (loc.text.isEmpty || price.text.isEmpty) return;
      final p = double.tryParse(price.text);
      if (p == null) return;
      InheritedStore.of(context).addMarketEntry(loc.text, p);
      loc.clear();
      price.clear();
      },
      child: const Text("যোগ"),
      )
      ],
      );
      }
      }

      class CommunityScreen extends StatefulWidget {
      const CommunityScreen({super.key});
      @override
      State<CommunityScreen> createState() => _CommunityScreenState();
      }
      class _CommunityScreenState extends State<CommunityScreen> {
      final q = TextEditingController();
      final a = TextEditingController();
      @override
      Widget build(BuildContext context) {
      final store = InheritedStore.of(context);
      final role = store.currentRole;
      return Column(
      children: [
      // Only farmers and stakeholders can post questions (as requirement), experts answer.
      if (role == UserRole.farmer || role == UserRole.stakeholder)
      Padding(
      padding: const EdgeInsets.all(12),
      child: Row(
      children: [
      Expanded(child: TextField(controller: q, decoration: const InputDecoration(hintText: "প্রশ্ন লিখুন…"))),
      const SizedBox(width: 8),
      ElevatedButton(
      onPressed: () {
      final text = q.text.trim();
      if (text.isEmpty) return;
      InheritedStore.of(context).addQuestion(text);
      q.clear();
      },
      child: const Text("পোস্ট"),
      )
      ],
      ),
      ),
      Expanded(
      child: ListView.builder(
      itemCount: store.qna.length,
      itemBuilder: (_, i) {
      final item = store.qna[i];
      return Card(
      child: ListTile(
      title: Text("প্রশ্ন: ${item.question}"),
      subtitle: Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
      const SizedBox(height: 6),
      Text(item.answer == null ? "উত্তর অপেক্ষমাণ…" : "উত্তর: ${item.answer}"),
      const SizedBox(height: 8),
      Text("প্রশ্নকারী: ${item.authorName} (${item.authorZone}) • ${item.authorRole}"),
      ],
      ),
      ),
      );
      },
      ),
      ),
      if (role == UserRole.expert)
      Padding(
      padding: const EdgeInsets.all(12),
      child: Row(
      children: [
      Expanded(child: TextField(controller: a, decoration: const InputDecoration(hintText: "উত্তর দিন…"))),
      const SizedBox(width: 8),
      ElevatedButton(
      onPressed: () {
      if (a.text.trim().isEmpty) return;
      final idx = InheritedStore.of(context).qna.indexWhere((e) => e.answer == null);
      if (idx != -1) {
      InheritedStore.of(context).answerQuestion(idx, a.text.trim());
      a.clear();
      }
      },
      child: const Text("সেন্ড"),
      )
      ],
      ),
      ),
      ],
      );
      }
      }
